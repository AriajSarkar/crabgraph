1//! Streaming AEAD encryption for large files.
2//!
3//! This module provides streaming encryption and decryption capabilities for processing
4//! large files that don't fit in memory. It uses the STREAM construction from
5//! [Online Authenticated-Encryption and its Nonce-Reuse Misuse-Resistance](https://eprint.iacr.org/2015/189.pdf)
6//! which provides nonce-reuse resistance.
7//!
8//! # Security Notes
9//!
10//! - Each chunk is independently authenticated
11//! - The stream uses a nonce derivation function to ensure unique nonces per chunk
12//! - The final chunk includes authentication of the entire stream
13//! - Maximum file size depends on chunk counter size (default: 2^32 chunks)
14//!
15//! # Example
16//!
17//! ```
18//! use crabgraph::aead::stream::{Aes256GcmStreamEncryptor, Aes256GcmStreamDecryptor};
19//! use crabgraph::CrabResult;
20//!
21//! # fn main() -> CrabResult<()> {
22//! // Generate a 32-byte key
23//! let key = crabgraph::rand::secure_bytes(32)?;
24//!
25//! // Encrypt data in chunks
26//! let plaintext = vec![b"Hello, world! ".repeat(1000)];
27//! let mut encryptor = Aes256GcmStreamEncryptor::new(&key)?;
28//! let nonce = encryptor.nonce();
29//! let mut ciphertext_chunks = Vec::new();
30//!
31//! // Encrypt all but the last chunk
32//! for chunk in &plaintext[..plaintext.len() - 1] {
33//!     ciphertext_chunks.push(encryptor.encrypt_next(chunk)?);
34//! }
35//! // Encrypt the last chunk (this consumes the encryptor)
36//! let last_chunk = &plaintext[plaintext.len() - 1];
37//! ciphertext_chunks.push(encryptor.encrypt_last(last_chunk)?);
38//!
39//! // Decrypt data in chunks
40//! let mut decryptor = Aes256GcmStreamDecryptor::from_nonce(&key, &nonce)?;
41//! let mut decrypted = Vec::new();
42//!
43//! // Decrypt all but the last chunk
44//! for chunk in &ciphertext_chunks[..ciphertext_chunks.len() - 1] {
45//!     decrypted.extend_from_slice(&decryptor.decrypt_next(chunk)?);
46//! }
47//! // Decrypt the last chunk (this consumes the decryptor)
48//! let last_encrypted = &ciphertext_chunks[ciphertext_chunks.len() - 1];
49//! decrypted.extend_from_slice(&decryptor.decrypt_last(last_encrypted)?);
50//!
51//! // Verify the decrypted data matches the first (and only) chunk in plaintext
52//! assert_eq!(&decrypted, plaintext[0].as_slice());
53//! # Ok(())
54//! # }
55//! ```

76/// Streaming encryptor for AES-256-GCM.
77///
78/// Accepts raw 32-byte keys and produces a stream of encrypted chunks.
79pub struct Aes256GcmStreamEncryptor {
80    encryptor: EncryptorBE32<Aes256Gcm>,
81    nonce: Vec<u8>,
82}
83
84impl Aes256GcmStreamEncryptor {
85    /// Creates a new streaming encryptor from a 32-byte key.
86    ///
87    /// # Arguments
88    ///
89    /// * `key` - A 32-byte AES-256 key
90    ///
91    /// # Errors
92    ///
93    /// Returns an error if the key is not exactly 32 bytes.
94    pub fn new(key: &[u8]) -> CrabResult<Self> {
95        if key.len() != 32 {
96            return Err(CrabError::invalid_input(format!(
97                "AES-256 requires 32-byte key, got {}",
98                key.len()
99            )));
100        }
101
102        // Generate a 7-byte nonce for STREAM (AES-GCM needs 12 bytes, STREAM uses 5 for counter+flag)
103        let nonce_bytes = crate::rand::secure_bytes(7)?;
104        let nonce_array: &AesStreamNonce = nonce_bytes.as_slice().into();
105
106        let key_array: &Key<Aes256Gcm> = key.into();
107        let aead = Aes256Gcm::new(key_array);
108        let encryptor = EncryptorBE32::from_aead(aead, nonce_array);
109
110        Ok(Self {
111            encryptor,
112            nonce: nonce_bytes,
113        })
114    }
115
116    /// Returns the nonce used for this stream.
117    ///
118    /// This nonce must be stored with the encrypted data to enable decryption.
119    pub fn nonce(&self) -> Vec<u8> {
120        self.nonce.clone()
121    }
122
123    /// Encrypts the next chunk of data.
124    ///
125    /// # Arguments
126    ///
127    /// * `plaintext` - The chunk of data to encrypt
128    pub fn encrypt_next(&mut self, plaintext: &[u8]) -> CrabResult<Vec<u8>> {
129        self.encryptor
130            .encrypt_next(plaintext)
131            .map_err(|e| CrabError::crypto_error(format!("Stream encryption failed: {}", e)))
132    }
133
134    /// Encrypts the last chunk and finalizes the stream.
135    ///
136    /// After calling this method, the encryptor cannot be used anymore.
137    ///
138    /// # Arguments
139    ///
140    /// * `plaintext` - The final chunk of data to encrypt (can be empty)
141    pub fn encrypt_last(self, plaintext: &[u8]) -> CrabResult<Vec<u8>> {
142        self.encryptor
143            .encrypt_last(plaintext)
144            .map_err(|e| CrabError::crypto_error(format!("Stream encryption failed: {}", e)))
145    }
146}

155    ///
156    /// # Arguments
157    ///
158    /// * `key` - A 32-byte AES-256 key (must match the encryption key)
159    /// * `nonce` - The 7-byte nonce from the encryption process

161    /// # Errors
162    ///

165        if key.len() != 32 {
166            return Err(CrabError::invalid_input(format!(

368    #[test]
369    fn test_aes256gcm_stream_encrypt_decrypt() -> CrabResult<()> {
370        let key = crate::rand::secure_bytes(32)?;
371        let plaintext_chunks = vec![b"Hello, ".as_slice(), b"world!".as_slice()];
372
373        // Encrypt
374        let mut encryptor = Aes256GcmStreamEncryptor::new(&key)?;
375        let nonce = encryptor.nonce();
376
377        let mut ciphertext_chunks = Vec::new();
378        for chunk in &plaintext_chunks {
379            ciphertext_chunks.push(encryptor.encrypt_next(chunk)?);
380        }
381        ciphertext_chunks.push(encryptor.encrypt_last(b"")?);
382
383        // Decrypt
384        let mut decryptor = Aes256GcmStreamDecryptor::from_nonce(&key, &nonce)?;
385        let mut decrypted = Vec::new();
386
387        let last_idx = ciphertext_chunks.len() - 1;
388        for (i, chunk) in ciphertext_chunks.iter().enumerate() {
389            if i == last_idx {
390                decrypted.extend_from_slice(&decryptor.decrypt_last(chunk)?);
391                break; // Decryptor is consumed, no more iterations
392            } else {
393                decrypted.extend_from_slice(&decryptor.decrypt_next(chunk)?);
394            }
395        }
396
397        let expected: Vec<u8> = plaintext_chunks.concat();
398        assert_eq!(decrypted, expected);
399
400        Ok(())
401    }

439    #[test]
440    fn test_stream_large_data() -> CrabResult<()> {
441        let key = crate::rand::secure_bytes(32)?;
442        let chunk_size = 1024;
443        let num_chunks = 100;
444
445        // Generate large plaintext
446        let mut plaintext = Vec::new();
447        for i in 0..num_chunks {
448            plaintext.extend_from_slice(&vec![i as u8; chunk_size]);
449        }
450
451        // Encrypt in chunks
452        let mut encryptor = Aes256GcmStreamEncryptor::new(&key)?;
453        let nonce = encryptor.nonce();
454
455        let mut ciphertext_chunks = Vec::new();
456        for chunk in plaintext.chunks(chunk_size) {
457            ciphertext_chunks.push(encryptor.encrypt_next(chunk)?);
458        }
459        ciphertext_chunks.push(encryptor.encrypt_last(&[])?);
460
461        // Decrypt in chunks
462        let mut decryptor = Aes256GcmStreamDecryptor::from_nonce(&key, &nonce)?;
463        let mut decrypted = Vec::new();
464
465        let last_idx = ciphertext_chunks.len() - 1;
466        for (i, chunk) in ciphertext_chunks.iter().enumerate() {
467            if i == last_idx {
468                decrypted.extend_from_slice(&decryptor.decrypt_last(chunk)?);
469                break; // Decryptor is consumed, no more iterations

472            }
473        }

480    #[test]
481    fn test_stream_authentication_failure() -> CrabResult<()> {
482        let key = crate::rand::secure_bytes(32)?;
483
484        // Encrypt
485        let mut encryptor = Aes256GcmStreamEncryptor::new(&key)?;
486        let nonce = encryptor.nonce();
487
488        let chunk1 = encryptor.encrypt_next(b"Hello")?;
489        let mut chunk2 = encryptor.encrypt_last(b"World")?;
490
491        // Corrupt the second chunk
492        chunk2[0] ^= 1;
493
494        // Decrypt
495        let mut decryptor = Aes256GcmStreamDecryptor::from_nonce(&key, &nonce)?;
496        decryptor.decrypt_next(&chunk1)?;
497
498        // Should fail authentication
499        let result = decryptor.decrypt_last(&chunk2);
500        assert!(result.is_err());
501
502        Ok(())
503    }
