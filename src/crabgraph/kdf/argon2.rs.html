1//! Argon2 key derivation.
2//!
3//! Argon2 is the recommended password hashing algorithm, winner of the
4//! Password Hashing Competition (2015). It provides strong resistance
5//! to GPU and ASIC attacks through memory-hard operations.
6
7use crate::errors::{CrabError, CrabResult};
8use crate::secrets::SecretVec;
9use argon2::{Algorithm, Argon2, Params, Version};
10
11/// Argon2 parameters configuration.
12///
13/// These parameters control the time and memory cost of the algorithm.
14/// Higher values provide better security but slower performance.
15#[derive(Debug, Clone)]
16pub struct Argon2Params {
17    /// Memory size in KiB (recommended: 64 MiB = 65536 KiB for interactive use)
18    pub memory_cost: u32,
19    /// Number of iterations (recommended: 3-4 for interactive use)
20    pub time_cost: u32,
21    /// Degree of parallelism (recommended: match CPU cores, e.g., 4)
22    pub parallelism: u32,
23}
24
25impl Default for Argon2Params {
26    /// Returns recommended parameters for interactive use (2025).
27    ///
28    /// These values balance security and user experience for authentication:
29    /// - 64 MiB memory
30    /// - 3 iterations
31    /// - 4 parallel threads
32    fn default() -> Self {
33        Self {
34            memory_cost: 65536, // 64 MiB
35            time_cost: 3,
36            parallelism: 4,
37        }
38    }
39}
40
41impl Argon2Params {
42    /// Creates parameters optimized for interactive authentication.
43    ///
44    /// Fast enough for login flows (~100-200ms on modern hardware).
45    pub fn interactive() -> Self {
46        Self::default()
47    }
48
49    /// Creates parameters optimized for high-security scenarios.
50    ///
51    /// Slower but provides maximum protection (~1-2 seconds).
52    pub fn high_security() -> Self {
53        Self {
54            memory_cost: 262144, // 256 MiB
55            time_cost: 5,
56            parallelism: 4,
57        }
58    }
59
60    /// Creates parameters optimized for low-memory environments.
61    ///
62    /// Suitable for embedded systems or constrained environments.
63    pub fn low_memory() -> Self {
64        Self {
65            memory_cost: 32768, // 32 MiB
66            time_cost: 4,
67            parallelism: 2,
68        }
69    }
70}

72/// Derives a key from a password using Argon2id with default parameters.
73///
74/// # Arguments
75/// * `password` - The password to derive from
76/// * `salt` - A unique salt (â‰¥16 bytes required)
77/// * `key_len` - Desired output key length in bytes
78///
79/// # Returns
80/// A `SecretVec` containing the derived key material
81///
82/// # Security Notes
83/// - Uses Argon2id variant (hybrid of Argon2i and Argon2d)
84/// - Default parameters: 64 MiB memory, 3 iterations, 4 parallelism
85/// - Salt must be at least 16 bytes and unique per password
86/// - Store the salt and parameters alongside the hash
87///
88/// # Example
89/// ```
90/// use crabgraph::kdf::argon2_derive;
91///
92/// let password = b"correct horse battery staple";
93/// let salt = b"unique_random_salt_16bytes!!";
94/// let key = argon2_derive(password, salt, 32).unwrap();
95///
96/// assert_eq!(key.len(), 32);
97/// ```
98pub fn argon2_derive(password: &[u8], salt: &[u8], key_len: usize) -> CrabResult<SecretVec> {
99    argon2_derive_with_params(password, salt, key_len, &Argon2Params::default())
100}

102/// Derives a key from a password using Argon2id with custom parameters.
103///
104/// # Arguments
105/// * `password` - The password to derive from
106/// * `salt` - A unique salt (â‰¥16 bytes required)
107/// * `key_len` - Desired output key length in bytes
108/// * `params` - Argon2 parameters (memory cost, time cost, parallelism)
109///
110/// # Example
111/// ```
112/// use crabgraph::kdf::{argon2_derive_with_params, Argon2Params};
113///
114/// let password = b"my_password";
115/// let salt = b"random_salt_16+b";
116/// let params = Argon2Params::high_security();
117/// let key = argon2_derive_with_params(password, salt, 32, &params).unwrap();
118///
119/// assert_eq!(key.len(), 32);
120/// ```
121pub fn argon2_derive_with_params(
122    password: &[u8],
123    salt: &[u8],
124    key_len: usize,
125    params: &Argon2Params,
126) -> CrabResult<SecretVec> {
127    if salt.len() < 16 {
128        return Err(CrabError::invalid_input("Argon2 requires salt of at least 16 bytes"));
129    }
130
131    if key_len == 0 || key_len > 4294967295 {
132        return Err(CrabError::invalid_input("Invalid key length"));
133    }
134
135    // Build Argon2 parameters
136    let argon2_params =
137        Params::new(params.memory_cost, params.time_cost, params.parallelism, Some(key_len))?;
138
139    let argon2 = Argon2::new(Algorithm::Argon2id, Version::V0x13, argon2_params);
140
141    // Derive key
142    let mut output = vec![0u8; key_len];
143    argon2
144        .hash_password_into(password, salt, &mut output)
145        .map_err(|e| CrabError::key_error(format!("Argon2 derivation failed: {}", e)))?;
146
147    Ok(SecretVec::new(output))
148}

150#[cfg(test)]
151mod tests {
152    use super::*;
153
154    #[test]
155    fn test_argon2_basic() {
156        let password = b"password123";
157        let salt = b"saltsaltsaltsalt";
158        let key = argon2_derive(password, salt, 32).unwrap();
159        assert_eq!(key.len(), 32);
160    }
161
162    #[test]
163    fn test_argon2_deterministic() {
164        let password = b"test_password";
165        let salt = b"test_salt_16byte";
166
167        let key1 = argon2_derive(password, salt, 32).unwrap();
168        let key2 = argon2_derive(password, salt, 32).unwrap();
169
170        assert_eq!(key1.as_slice(), key2.as_slice());
171    }
172
173    #[test]
174    fn test_argon2_different_passwords() {
175        let salt = b"same_salt_16byte";
176
177        let key1 = argon2_derive(b"password1", salt, 32).unwrap();
178        let key2 = argon2_derive(b"password2", salt, 32).unwrap();
179
180        assert_ne!(key1.as_slice(), key2.as_slice());
181    }
