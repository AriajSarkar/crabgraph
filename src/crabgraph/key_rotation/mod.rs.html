1//! Key rotation and versioning utilities.
2//!
3//! This module provides tools for managing cryptographic key rotation, a critical
4//! security practice where keys are periodically replaced to limit exposure from
5//! potential key compromise.
6//!
7//! # Features
8//!
9//! - **Key Versioning**: Track which version of a key encrypted each piece of data
10//! - **Multiple Key Storage**: Securely store multiple key versions simultaneously
11//! - **Automatic Re-encryption**: Utilities to decrypt with old keys and encrypt with new ones
12//! - **Zero-downtime Rotation**: Support for gradual key transitions without service interruption
13//!
14//! # Security Considerations
15//!
16//! - Old keys should be retained only as long as needed for re-encryption
17//! - Set a maximum number of key versions to prevent unbounded growth
18//! - Implement a key deletion policy after successful re-encryption
19//! - Audit key usage to ensure old keys are eventually phased out
20//! - Use secure key storage (HSM, KMS) for production deployments
21//!
22//! # Example
23//!
24//! ```
25//! use crabgraph::{
26//!     key_rotation::KeyRotationManager,
27//!     aead::{AesGcm256, CrabAead},
28//!     CrabResult,
29//! };
30//!
31//! fn rotation_example() -> CrabResult<()> {
32//!     // Create a key rotation manager
33//!     let mut manager = KeyRotationManager::<AesGcm256>::new()?;
34//!     
35//!     // Encrypt data with version 1 key
36//!     let data = b"Secret data";
37//!     let (version, ciphertext) = manager.encrypt(data, None)?;
38//!     assert_eq!(version, 1);
39//!     
40//!     // Rotate to a new key (version 2)
41//!     manager.rotate()?;
42//!     
43//!     // Can still decrypt old data with version 1
44//!     let decrypted = manager.decrypt(version, &ciphertext, None)?;
45//!     assert_eq!(decrypted, data);
46//!     
47//!     // Re-encrypt old data with the new key
48//!     let (new_version, new_ciphertext) = manager.re_encrypt(version, &ciphertext, None)?;
49//!     assert_eq!(new_version, 2);
50//!     
51//!     Ok(())
52//! }
53//! ```

63/// Trait for AEAD ciphers that support key generation and creation.
64///
65/// This extends `CrabAead` with methods needed for key rotation.
66pub trait RotatableAead: CrabAead {
67    /// Generate a new random key for this cipher.
68    fn generate_key() -> CrabResult<SecretVec>;
69
70    /// Create a new cipher instance from a key.
71    fn from_key(key: &SecretVec) -> CrabResult<Self>
72    where
73        Self: Sized;
74}

76/// A versioned cryptographic key.
77///
78/// Wraps a key with version information to track which version was used
79/// for encryption operations.
80#[derive(Clone, Zeroize)]
81#[zeroize(drop)]
82pub struct VersionedKey {
83    /// The version number (starts at 1)
84    pub version: u32,
85    /// The actual key material (auto-zeroized on drop)
86    pub key: SecretVec,
87}
88
89impl VersionedKey {
90    /// Create a new versioned key.
91    ///
92    /// # Arguments
93    ///
94    /// * `version` - The version number (must be >= 1)
95    /// * `key` - The key material
96    ///
97    /// # Errors
98    ///
99    /// Returns `CrabError::InvalidInput` if version is 0.
100    pub fn new(version: u32, key: SecretVec) -> CrabResult<Self> {
101        if version == 0 {
102            return Err(CrabError::invalid_input("Key version must be >= 1"));
103        }
104        Ok(Self { version, key })
105    }
106
107    /// Get the version number.
108    pub fn version(&self) -> u32 {
109        self.version
110    }
111
112    /// Get a reference to the key material.
113    pub fn key(&self) -> &SecretVec {
114        &self.key
115    }
116}

118/// Manager for cryptographic key rotation.
119///
120/// Maintains multiple versions of keys and provides utilities for encrypting with
121/// the current key, decrypting with any version, and re-encrypting data from old
122/// to new keys.
123///
124/// # Type Parameters
125///
126/// * `C` - The AEAD cipher type (e.g., `AesGcm256`, `ChaCha20Poly1305`)
127///
128/// # Security Notes
129///
130/// - Keys are stored in memory using `SecretVec` (auto-zeroized on drop)
131/// - Maximum 256 key versions by default (prevents unbounded memory growth)
132/// - Old keys should be removed after successful re-encryption
133/// - Consider using HSM/KMS for production key storage
134///
135/// # Example
136///
137/// ```
138/// use crabgraph::{
139///     key_rotation::KeyRotationManager,
140///     aead::ChaCha20Poly1305,
141///     CrabResult,
142/// };
143///
144/// fn example() -> CrabResult<()> {
145///     let mut manager = KeyRotationManager::<ChaCha20Poly1305>::new()?;
146///     
147///     // Encrypt with current key (version 1)
148///     let (v1, ct1) = manager.encrypt(b"data1", None)?;
149///     
150///     // Rotate to version 2
151///     manager.rotate()?;
152///     let (v2, ct2) = manager.encrypt(b"data2", None)?;
153///     
154///     // Both versions can decrypt their data
155///     assert_eq!(manager.decrypt(v1, &ct1, None)?, b"data1");
156///     assert_eq!(manager.decrypt(v2, &ct2, None)?, b"data2");
157///     
158///     // Re-encrypt old data with new key
159///     let (v2_new, ct1_new) = manager.re_encrypt(v1, &ct1, None)?;
160///     assert_eq!(v2_new, v2);
161///     
162///     Ok(())
163/// }
164/// ```
165pub struct KeyRotationManager<C: RotatableAead> {
166    /// All key versions (version number -> key)
167    keys: HashMap<u32, SecretVec>,
168    /// Current (latest) key version
169    current_version: u32,
170    /// Maximum number of versions to keep
171    max_versions: usize,
172    /// Phantom data for the cipher type
173    _phantom: std::marker::PhantomData<C>,
174}
175
176impl<C: RotatableAead> KeyRotationManager<C> {
177    /// Create a new key rotation manager with an initial key.
178    ///
179    /// Generates a random key as version 1.
180    ///
181    /// # Errors
182    ///
183    /// Returns an error if key generation fails.
184    ///
185    /// # Example
186    ///
187    /// ```
188    /// use crabgraph::{key_rotation::KeyRotationManager, aead::AesGcm256, CrabResult};
189    ///
190    /// fn example() -> CrabResult<()> {
191    ///     let manager = KeyRotationManager::<AesGcm256>::new()?;
192    ///     assert_eq!(manager.current_version(), 1);
193    ///     Ok(())
194    /// }
195    /// ```
196    pub fn new() -> CrabResult<Self> {
197        Self::with_max_versions(256)
198    }
199
200    /// Create a new key rotation manager with a specified maximum number of versions.
201    ///
202    /// # Arguments
203    ///
204    /// * `max_versions` - Maximum number of key versions to retain (must be >= 1)
205    ///
206    /// # Errors
207    ///
208    /// Returns `CrabError::InvalidInput` if `max_versions` is 0.
209    ///
210    /// # Example
211    ///
212    /// ```
213    /// use crabgraph::{key_rotation::KeyRotationManager, aead::AesGcm256, CrabResult};
214    ///
215    /// fn example() -> CrabResult<()> {
216    ///     // Keep only last 10 key versions
217    ///     let manager = KeyRotationManager::<AesGcm256>::with_max_versions(10)?;
218    ///     Ok(())
219    /// }
220    /// ```
221    pub fn with_max_versions(max_versions: usize) -> CrabResult<Self> {
222        if max_versions == 0 {
223            return Err(CrabError::invalid_input("max_versions must be >= 1"));
224        }
225
226        let key = C::generate_key()?;
227        let mut keys = HashMap::new();
228        keys.insert(1, key);
229
230        Ok(Self {
231            keys,
232            current_version: 1,
233            max_versions,
234            _phantom: std::marker::PhantomData,
235        })
236    }
237
238    /// Create a manager from an existing key as version 1.
239    ///
240    /// Useful when you have an existing key to manage.
241    ///
242    /// # Arguments
243    ///
244    /// * `key` - The initial key
245    ///
246    /// # Example
247    ///
248    /// ```
249    /// use crabgraph::{
250    ///     key_rotation::KeyRotationManager,
251    ///     aead::{AesGcm256, CrabAead},
252    ///     secrets::SecretVec,
253    ///     CrabResult,
254    /// };
255    ///
256    /// fn example() -> CrabResult<()> {
257    ///     let existing_key_bytes = AesGcm256::generate_key()?;
258    ///     let existing_key = SecretVec::new(existing_key_bytes);
259    ///     let manager = KeyRotationManager::<AesGcm256>::from_key(existing_key)?;
260    ///     Ok(())
261    /// }
262    /// ```
263    pub fn from_key(key: SecretVec) -> CrabResult<Self> {
264        let mut keys = HashMap::new();
265        keys.insert(1, key);
266
267        Ok(Self {
268            keys,
269            current_version: 1,
270            max_versions: 256,
271            _phantom: std::marker::PhantomData,
272        })
273    }

275    /// Rotate to a new key version.
276    ///
277    /// Generates a new random key and increments the version number.
278    /// Old keys are retained for decryption.
279    ///
280    /// If the number of versions exceeds `max_versions`, the oldest key is removed.
281    ///
282    /// # Errors
283    ///
284    /// Returns an error if key generation fails or if version would overflow.
285    ///
286    /// # Example
287    ///
288    /// ```
289    /// use crabgraph::{key_rotation::KeyRotationManager, aead::AesGcm256, CrabResult};
290    ///
291    /// fn example() -> CrabResult<()> {
292    ///     let mut manager = KeyRotationManager::<AesGcm256>::new()?;
293    ///     assert_eq!(manager.current_version(), 1);
294    ///     
295    ///     manager.rotate()?;
296    ///     assert_eq!(manager.current_version(), 2);
297    ///     
298    ///     Ok(())
299    /// }
300    /// ```
301    pub fn rotate(&mut self) -> CrabResult<()> {
302        let new_version = self
303            .current_version
304            .checked_add(1)
305            .ok_or_else(|| CrabError::invalid_input("Version number overflow"))?;
306
307        let new_key = C::generate_key()?;
308        self.keys.insert(new_version, new_key);
309        self.current_version = new_version;
310
311        // Remove oldest key if exceeding max_versions
312        if self.keys.len() > self.max_versions {
313            let oldest_version = self.current_version - self.max_versions as u32;
314            self.keys.remove(&oldest_version);
315        }
316
317        Ok(())
318    }
319
320    /// Rotate to a new key version using a provided key.
321    ///
322    /// Similar to `rotate()`, but uses a specific key instead of generating a random one.
323    ///
324    /// # Arguments
325    ///
326    /// * `key` - The new key to use
327    ///
328    /// # Errors
329    ///
330    /// Returns an error if version would overflow.
331    pub fn rotate_with_key(&mut self, key: SecretVec) -> CrabResult<()> {
332        let new_version = self
333            .current_version
334            .checked_add(1)
335            .ok_or_else(|| CrabError::invalid_input("Version number overflow"))?;
336
337        self.keys.insert(new_version, key);
338        self.current_version = new_version;
339
340        // Remove oldest key if exceeding max_versions
341        if self.keys.len() > self.max_versions {
342            let oldest_version = self.current_version - self.max_versions as u32;
343            self.keys.remove(&oldest_version);
344        }
345
346        Ok(())
347    }

349    /// Get the current (latest) key version number.
350    ///
351    /// # Example
352    ///
353    /// ```
354    /// use crabgraph::{key_rotation::KeyRotationManager, aead::AesGcm256, CrabResult};
355    ///
356    /// fn example() -> CrabResult<()> {
357    ///     let manager = KeyRotationManager::<AesGcm256>::new()?;
358    ///     println!("Current version: {}", manager.current_version());

361    /// ```
362    pub fn current_version(&self) -> u32 {
363        self.current_version
364    }

367    pub fn version_count(&self) -> usize {

371    /// Check if a specific key version exists.
372    ///
373    /// # Arguments
374    ///
375    /// * `version` - The version to check
376    pub fn has_version(&self, version: u32) -> bool {
377        self.keys.contains_key(&version)
378    }

380    /// Encrypt data with the current (latest) key version.
381    ///
382    /// Returns the version number and ciphertext.
383    ///
384    /// # Arguments
385    ///
386    /// * `plaintext` - Data to encrypt
387    /// * `aad` - Optional associated data (authenticated but not encrypted)
388    ///
389    /// # Returns
390    ///
391    /// A tuple of `(version, ciphertext)` where version is the key version used.
392    ///
393    /// # Errors
394    ///
395    /// Returns an error if encryption fails.
396    ///
397    /// # Example
398    ///
399    /// ```
400    /// use crabgraph::{key_rotation::KeyRotationManager, aead::AesGcm256, CrabResult};
401    ///
402    /// fn example() -> CrabResult<()> {
403    ///     let manager = KeyRotationManager::<AesGcm256>::new()?;
404    ///     let (version, ciphertext) = manager.encrypt(b"secret data", None)?;
405    ///     println!("Encrypted with key version {}", version);
406    ///     Ok(())
407    /// }
408    /// ```
409    pub fn encrypt(&self, plaintext: &[u8], aad: Option<&[u8]>) -> CrabResult<(u32, Ciphertext)> {
410        let key = self
411            .keys
412            .get(&self.current_version)
413            .ok_or_else(|| CrabError::invalid_input("Current key version not found"))?;
414
415        let cipher = C::from_key(key)?;
416        let ciphertext = cipher.encrypt(plaintext, aad)?;
417
418        Ok((self.current_version, ciphertext))
419    }
420
421    /// Decrypt data using a specific key version.
422    ///
423    /// # Arguments
424    ///
425    /// * `version` - The key version to use for decryption
426    /// * `ciphertext` - The encrypted data
427    /// * `aad` - Optional associated data (must match what was used during encryption)
428    ///
429    /// # Errors
430    ///
431    /// Returns an error if:
432    /// - The specified version doesn't exist
433    /// - Decryption fails (wrong key, tampered data, etc.)
434    ///
435    /// # Example
436    ///
437    /// ```
438    /// use crabgraph::{key_rotation::KeyRotationManager, aead::AesGcm256, CrabResult};
439    ///
440    /// fn example() -> CrabResult<()> {
441    ///     let manager = KeyRotationManager::<AesGcm256>::new()?;
442    ///     let (version, ciphertext) = manager.encrypt(b"secret", None)?;
443    ///     let plaintext = manager.decrypt(version, &ciphertext, None)?;
444    ///     assert_eq!(plaintext, b"secret");
445    ///     Ok(())
446    /// }
447    /// ```
448    pub fn decrypt(
449        &self,
450        version: u32,
451        ciphertext: &Ciphertext,
452        aad: Option<&[u8]>,
453    ) -> CrabResult<Vec<u8>> {
454        let key = self.keys.get(&version).ok_or_else(|| {
455            CrabError::invalid_input(format!("Key version {} not found", version))
456        })?;
457
458        let cipher = C::from_key(key)?;
459        cipher.decrypt(ciphertext, aad)
460    }

462    /// Re-encrypt data from an old key version to the current version.
463    ///
464    /// This is the core utility for key rotation: it decrypts data with an old key
465    /// and re-encrypts it with the current key in a single operation.
466    ///
467    /// # Arguments
468    ///
469    /// * `old_version` - The version that was used to encrypt the data
470    /// * `old_ciphertext` - The data encrypted with the old key
471    /// * `aad` - Optional associated data (must match original encryption)
472    ///
473    /// # Returns
474    ///
475    /// A tuple of `(new_version, new_ciphertext)` encrypted with the current key.
476    ///
477    /// # Errors
478    ///
479    /// Returns an error if:
480    /// - The old version doesn't exist
481    /// - Decryption with old key fails
482    /// - Encryption with new key fails
483    ///
484    /// # Example
485    ///
486    /// ```
487    /// use crabgraph::{key_rotation::KeyRotationManager, aead::AesGcm256, CrabResult};
488    ///
489    /// fn example() -> CrabResult<()> {
490    ///     let mut manager = KeyRotationManager::<AesGcm256>::new()?;
491    ///     
492    ///     // Encrypt with version 1
493    ///     let (v1, ct1) = manager.encrypt(b"data", None)?;
494    ///     
495    ///     // Rotate to version 2
496    ///     manager.rotate()?;
497    ///     
498    ///     // Re-encrypt from v1 to v2
499    ///     let (v2, ct2) = manager.re_encrypt(v1, &ct1, None)?;
500    ///     assert_eq!(v2, 2);
501    ///     
502    ///     // Verify new ciphertext decrypts correctly
503    ///     let plaintext = manager.decrypt(v2, &ct2, None)?;
504    ///     assert_eq!(plaintext, b"data");
505    ///     
506    ///     Ok(())
507    /// }
508    /// ```
509    pub fn re_encrypt(
510        &self,
511        old_version: u32,
512        old_ciphertext: &Ciphertext,
513        aad: Option<&[u8]>,
514    ) -> CrabResult<(u32, Ciphertext)> {
515        // Decrypt with old key
516        let plaintext = self.decrypt(old_version, old_ciphertext, aad)?;
517
518        // Encrypt with current key
519        self.encrypt(&plaintext, aad)
520    }
521
522    /// Remove a specific key version.
523    ///
524    /// Use this after successfully re-encrypting all data that used this version.
525    ///
526    /// # Arguments
527    ///
528    /// * `version` - The version to remove
529    ///
530    /// # Errors
531    ///
532    /// Returns an error if trying to remove the current version.
533    ///
534    /// # Security Note
535    ///
536    /// Ensure all data encrypted with this version has been re-encrypted before removal,
537    /// otherwise it will become unrecoverable.
538    ///
539    /// # Example
540    ///
541    /// ```
542    /// use crabgraph::{key_rotation::KeyRotationManager, aead::AesGcm256, CrabResult};
543    ///
544    /// fn example() -> CrabResult<()> {
545    ///     let mut manager = KeyRotationManager::<AesGcm256>::new()?;
546    ///     manager.rotate()?;  // Now at version 2
547    ///     
548    ///     // After re-encrypting all v1 data...
549    ///     manager.remove_version(1)?;
550    ///     
551    ///     assert!(!manager.has_version(1));
552    ///     assert!(manager.has_version(2));
553    ///     
554    ///     Ok(())
555    /// }
556    /// ```
557    pub fn remove_version(&mut self, version: u32) -> CrabResult<()> {
558        if version == self.current_version {
559            return Err(CrabError::invalid_input("Cannot remove current key version"));
560        }
561
562        self.keys.remove(&version);
563        Ok(())
564    }
565
566    /// Get a list of all key versions currently available.
567    ///
568    /// Returns versions in ascending order.
569    ///
570    /// # Example
571    ///
572    /// ```
573    /// use crabgraph::{key_rotation::KeyRotationManager, aead::AesGcm256, CrabResult};
574    ///
575    /// fn example() -> CrabResult<()> {
576    ///     let mut manager = KeyRotationManager::<AesGcm256>::new()?;
577    ///     manager.rotate()?;
578    ///     manager.rotate()?;
579    ///     
580    ///     let versions = manager.available_versions();
581    ///     assert_eq!(versions, vec![1, 2, 3]);
582    ///     
583    ///     Ok(())
584    /// }
585    /// ```
586    pub fn available_versions(&self) -> Vec<u32> {
587        let mut versions: Vec<u32> = self.keys.keys().copied().collect();
588        versions.sort_unstable();
589        versions
590    }
591}

594impl<C: RotatableAead> Drop for KeyRotationManager<C> {

597        self.keys.clear();
598    }

852        let mut plaintexts = vec![];
853        let mut ciphertexts = vec![];

861            ciphertexts.push((version, ciphertext));
862
863            if i < 4 {
864                manager.rotate().unwrap();
865            }
866        }
867
868        // All should decrypt correctly
869        for (i, (version, ciphertext)) in ciphertexts.iter().enumerate() {
870            let decrypted = manager.decrypt(*version, ciphertext, None).unwrap();
871            assert_eq!(decrypted, plaintexts[i].as_bytes());
872        }
